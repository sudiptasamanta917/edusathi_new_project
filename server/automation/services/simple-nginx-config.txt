server {
    listen 80;
    server_name test.thecomputeracademy.org;

    root /var/www/the-computer/center/dist;
    index index.html;

    # Static files cache
    location ~* \.(js|css|png|jpg|jpeg|gif|ico|svg|woff|woff2|ttf|eot)$ {
        expires 1y;
        add_header Cache-Control "public, immutable";
    }

    # Assets
    location /assets/ {
        alias /var/www/the-computer/center/dist/assets/;
    }

    # Uploads
    location ^~ /uploads/ {
        alias /var/www/the-computer/server/public/uploads/;
        autoindex on;
        try_files $uri $uri/ =404;
    }

    # API proxy
    location /api {
        proxy_pass http://localhost:4074;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    # SPA fallback
    location / {
        try_files $uri $uri/ /index.html =404;
    }
}
